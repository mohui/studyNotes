```javascript
/**
 * 百元买百鸡 公鸡5,母鸡3,小鸡三只1
 *
 * 公鸡最多买100 / 5 = 20,
 * 母鸡最多买 100 / 3 = 33
 * 小鸡最多买 300
 */
for (let i = 0; i < 20; i++) {
  for (let j = 0; j < 33; j++) {
    const q = 100 - i - j;
    if (i + j + q === 100 && 5 * i + 3 * j + q / 3 === 100) {
      console.log({i, j, q});
    }
  }
}
```
