export default class Evaluation {
    async test(area) {
        const childTree = await getAreaTree(area);
        // 过滤出所有的中心机构和卫生院
        const areaList = childTree.filter(
            it =>
                it.label == AreaLabel.CENTRE_CENTER || it.label == AreaLabel.CENTRE_HOME
        );
        // 取出所有地区ID
        const areaIdList = areaList.map(it => it.code);
        for (const it of areaIdList) {
            const a = await originalDB.execute(
                `
      select code from area where parent = ?
      `,
                it
            );
            const idList = a.map(mapIt => mapIt.code);
            const staffs = await appDB.execute(
                `
            select 1
            from staff_area_mapping
            where area in (${idList.map(_ => '?').join()})`,
                ...idList
            );
            if (staffs.length == 0) {
                console.log(it);
            }
        }
    }
}
