## 创建表
```sql
create table 表名(
    字段1 类型 [comment 注释]
)[comment 表注释]
```
### unsigned 无符号
- 如年龄
```sql
create table test (
    age tinyint unsigned comment '年龄'
)
```
### 添加字段
```sql
ALTER TABLE 表名 
ADD COLUMN 字段名 tinyint NOT NULL DEFAULT 0 COMMENT '注释' AFTER `match_fail_msg`;
```


## 删除表
```sql
drop table [if exists] 表名
```
## 删除指定表, 并重新创建该表
```sql
truncate table 表名;
```

### 删除字段名
```sql
alter table 表名 drop 字段名
```

## 表创建示例
```sql
CREATE TABLE IF NOT EXISTS log_metrics
(
    id           BIGINT UNSIGNED NOT NULL COMMENT '唯一标识',
    metrics_type VARCHAR(16)     NOT NULL COMMENT '指标类型',
    user_id      BIGINT UNSIGNED NOT NULL COMMENT '用户唯一id',
    extend_value JSON            NOT NULL COMMENT '扩展信息',
    origin_ip    VARCHAR(16)     NOT NULL DEFAULT '' COMMENT '来源ip',
    user_agent   VARCHAR(500)    NOT NULL DEFAULT '' COMMENT '客户端信息',
    created_by   BIGINT UNSIGNED NULL COMMENT '创建人',
    created_at   TIMESTAMP(3)    NOT NULL DEFAULT current_timestamp(3) COMMENT '创建时间',
    PRIMARY KEY (metrics_type, id)
)
    ENGINE = InnoDB
    CHARACTER SET = utf8mb4
    COLLATE = utf8mb4_bin
    COMMENT = '日志度量表（流水记录）'
    PARTITION BY LIST COLUMNS (metrics_type)(
        PARTITION user_purchase VALUES IN ("USER_PURCHASE"),
        PARTITION user_login VALUES IN ("USER_LOGIN"),
        PARTITION user_share VALUES IN ("USER_SHARE"),
        PARTITION show_share VALUES IN ("SHOW_SHARE")
        );
```
